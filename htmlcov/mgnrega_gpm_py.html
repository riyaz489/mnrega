318 old_request_data[3] == self.project_id:

324 request_msg += str(name[0]) + "|" + self.member_id + "|" + self.project_id+'|'+str(project_name[0]) 

325 self.conn.register_complain(str(RequestType.APPROVAL.name), self.gpm_id, request_msg, self.bdo_id)
